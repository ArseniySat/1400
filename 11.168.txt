heights = [8848, 6190, 5633, 5510, 5500, 5209, 5010, 4810, 4750, 4695, 4478, 4392, 4368, 4322, 4276, 4167, 4166, 4085, 4049, 4000]

# Пользовательский ввод для добавления новой высоты
new_height = int(input("Введите новую высоту: "))

# Пользовательский ввод для указания позиции добавления новой высоты
index = int(input("Введите индекс, после которого нужно вставить новую высоту: "))

# Используем операции со срезами списка для создания нового массива
# Создаем срез от начала исходного массива до позиции добавления новой высоты
first_part = heights[:index+1]
# Создаем срез от позиции добавления новой высоты до конца исходного массива
second_part = heights[index+1:]
# Создаем новый массив из первой части, новой высоты и второй части
new_heights = first_part + [new_height] + second_part

# Выводим новый массив
print(new_heights)
