x1_1 = int(input("Введите координату x левого нижнего угла первого прямоугольника: "))
y1_1 = int(input("Введите координату y левого нижнего угла первого прямоугольника: "))
x2_1 = int(input("Введите координату x правого нижнего угла первого прямоугольника: "))
y2_1 = int(input("Введите координату y правого нижнего угла первого прямоугольника: "))

# Координаты второго прямоугольника
x1_2 = int(input("Введите координату x левого нижнего угла второго прямоугольника: "))
y1_2 = int(input("Введите координату y левого нижнего угла второго прямоугольника: "))
x2_2 = int(input("Введите координату x правого нижнего угла второго прямоугольника: "))
y2_2 = int(input("Введите координату y правого нижнего угла второго прямоугольника: "))

# Находим минимальную координату x и y
min_x = min(x1_1, x2_1, x1_2, x2_2)
min_y = min(y1_1, y2_1, y1_2, y2_2)

# Находим максимальную координату x и y
max_x = max(x1_1, x2_1, x1_2, x2_2)
max_y = max(y1_1, y2_1, y1_2, y2_2)

print("Координаты левого нижнего угла минимального прямоугольника: ", min_x, min_y)
print("Координаты правого верхнего угла минимального прямоугольника: ", max_x, max_y)
