arr = [3, 3, 3, 5, 6, 7, 8, 9]
last_idx = arr.index(arr[0], 1)

# выводим все элементы, следующие за последним из равных элементов
if last_idx < len(arr):
    print("Количество равных элементов в начале массива:", last_idx)
    print("Следующие за последним из равных элементы:", arr[last_idx:])
else:
    print("Все элементы равны в начале массива")
