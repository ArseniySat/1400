# решение задачи с помощью лямбда-функции и функции map

cells = 1 # количество клеток в начале
hours = [3, 6, 9, 12, 15, 18, 21, 24] # список часов, для которых нужно вычислить количество клеток

cells_list = list(map(lambda h: 2**(h//3) * cells, hours)) # список количества клеток для каждого часа

for i in range(len(hours)):
    print("Через", hours[i], "часов будет", cells_list[i], "клеток") # выводим результат
