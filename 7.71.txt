exam1 = [5, 4, 4, 2, 3, 5, 4, 5, 3, 2]
exam2 = [3, 4, 5, 2, 4, 3, 2, 5, 2, 4]

# Используем функцию filter для нахождения студентов с двойкой на обоих экзаменах и подсчета их количества
count = len(list(filter(lambda x: x[0] == 2 and x[1] == 2, zip(exam1, exam2))))

print("Количество студентов, получивших на обоих экзаменах двойку:", count)
