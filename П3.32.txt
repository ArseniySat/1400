# а) Условие для треугольной фигуры:
x = float(input())
y = float(input())

if -2 <= x <= 2 and -4 <= y <= 1.5 and y + 2*x >= -5:
    print("Точка находится в заштрихованной области")
else:
    print("Точка не находится в заштрихованной области")


# б) Условие для прямоугольной фигуры:
x = float(input())
y = float(input())

if (y == 1.5 or y == -2) and -float('inf') <= x <= float('inf'):
    print("Точка находится в заштрихованной области")
elif -float('inf') <= x <= float('inf') and -2 < y < 1.5:
    print("Точка находится в заштрихованной области")
else:
    print("Точка не находится в заштрихованной области")


# в) Условие для прямоугольной фигуры:
x = float(input())
y = float(input())

if 1 <= x <= 2 and 1 <= y <= 4:
    print("Точка находится в заштрихованной области")
elif 1 <= x <= 2 and -float('inf') <= y <= 1:
    print("Точка находится в заштрихованной области")
else:
    print("Точка не находится в заштрихованной области")
