n = int(input("Введите число n (1 <= n <= 999): "))

# находим последнюю цифру числа х
last_digit = n % 10

# переставляем цифры в оставшемся двузначном числе
new_num = (n // 10) % 10 * 10 + (n // 100)

# добавляем последнюю цифру числа х в начало
new_num = last_digit * 100 + new_num

print("Число х:", new_num)
