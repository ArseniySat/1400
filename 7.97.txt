arr = [] # Создание пустого списка
num = int(input("Введите число: ")) # Ввод первого числа
i = 0 # Начальный индекс

while num != -1: # Чтение чисел до тех пор, пока не введен -1
    arr.append(num) # Добавление числа в список
    num = int(input("Введите число: ")) # Чтение следующего числа
    i += 1 # Приращение индекса

if any([arr[j] == arr[j+1] for j in range(len(arr)-1)]): # Проверка наличия одинаковых соседних чисел в списке
    print("Пара одинаковых соседних чисел найдена!")
    j = [j for j in range(len(arr)-1) if arr[j] == arr[j+1]][0] # Нахождение порядковых номеров первой из таких пар
    print("Их порядковые номера:", j+1, "и", j+2)
else:
    print("Пара одинаковых соседних чисел не найдена.")
