V = (4/3) * π * r^3
r1 = 5 + 5 # внешний радиус первого шара
r2 = r1 - 5 - 5 # внутренний радиус второго шара
r3 = r2 - 5 - 5 # внутренний радиус третьего шара и т.д.

V1 = (4/3) * 3.14 * r1**3 # объем первого шара
V2 = (4/3) * 3.14 * r2**3 # объем второго шара
V3 = (4/3) * 3.14 * r3**3 # объем третьего шара и т.д.

V_total = V1 + V2 + V3 # суммарный объем всех шаров в литрах

print("Суммарный объем всех шаров:", round(V_total/1000, 2), "л")
