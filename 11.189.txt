arr = [3,6,2,5,1,5,4,7,8,5,9] # пример массива

first_index = -1 # начальное значение индекса первого элемента 5
last_index = -1 # начальное значение индекса последнего элемента 5

for i in range(len(arr)):
    if arr[i] == 5 and first_index == -1: # если первое вхождение элемента
        first_index = i # сохраняем индекс
    elif arr[i] == 5: # если не первое вхождение элемента
        last_index = i # обновляем индекс

print("Первый индекс: ", first_index) # выводим результат
print("Последний индекс: ", last_index)
