d = [2, 4, 1, 7, 3, 9, 8, 6, 5, 10, 12, 17, 13, 14, 11, 15, 18, 19, 20, 16]

result = [(i, j) for i, j in enumerate(zip(d, d[1:]), 1) if i % 2 != 0 and j[0] % 2 != 0 and j[1] % 2 != 0]

if result:
    print("Со соседних нечётных чисел: ", result[0][1][0], result[0][1][1], " (порядковые номера: ", result[0][0], result[0][0]+1, ")")
else:
    print("В последовательности нет соседних нечётных чисел.")
