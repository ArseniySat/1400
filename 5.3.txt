# Решение задачи "столбиком" с помощью цикла for:

# а)
for i in range(20, 36):
    print(i)

# б)
a = int(input("Введите число a: "))
for i in range(a, 51):
    print(i**2)

# в)
b = int(input("Введите число b: "))
for i in range(10, b+1):
    print(i**3)

# г)
a = int(input("Введите число a: "))
b = int(input("Введите число b: "))
for i in range(a, b+1):
    print(i)
