cards = {
    7: 'шестерка',
    8: 'семерка',
    9: 'восьмерка',
    10: 'девятка',
    11: 'десятка',
    12: 'валет',
    13: 'дама',
    14: 'король',
    15: 'туз'
}

k = int(input('Введите номер карты от 7 до 14: '))
while k <= 6 or k > 14:
    print('Некорректный номер карты')
    k = int(input('Введите номер карты от 7 до 14: '))
    
for number, card in cards.items():
    if k == number:
        print(f'Достоинство карты: {card}')
        break
