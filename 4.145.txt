# создание списка соответствий между номером года и его названием

    "Красный Петух", "Черная Собака", "Красная Свинья"
]

# ввод номера года нашей эры
year = int(input("Введите номер года нашей эры: "))

# расчет номера года в цикле
cycle_year = (year - 4) % 60

# расчет номера подцикла
subcycle = cycle_year // 12

# расчет номера животного в подцикле
animal_number = cycle_year % 12

# расчет цвета животного
color = animal_number // 2

# вывод названия года
