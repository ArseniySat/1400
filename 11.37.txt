arr = [1.5, -2.6, 0, 5.9, -7.3, 0, 8.2, 3.7, -4.9] #исходный массив

#а) Ко всем отрицательным элементам прибавляем элемент с номером a1, из всех нулевых вычитаем число b
a1 = arr[0] #элемент с номером a1
b = 0 #задаем число b
arr = list(map(lambda x: x + a1 if x < 0 else x - b if x == 0 else x, arr))

print("Массив после преобразования а:\n", arr)

#б) Из всех положительных элементов вычитаем а, из всех отрицательных вычитаем b, ко всем нулевым прибавляем с
a = 2.5 #задаем число а
b = -3.5 #задаем число b
c = 10 #задаем число с
arr = list(map(lambda x: x - a if x > 0 else x - b if x < 0 else x + c, arr))

print("Массив после преобразования б:\n", arr)
