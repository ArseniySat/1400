m = list(map(int, input("Введите целые числа m через пробел: ").split()))

sum_1 = sum(filter(lambda x: x < 25.5, m))
sum_2 = sum(filter(lambda x: x <= 20, m))

if sum_1 <= 50:
    print("Сумма чисел меньше 25.5 не превышает 50")
else:
    print("Сумма чисел меньше 25.5 превышает 50")
    
if sum_2 % 3 == 0:
    print("Сумма чисел не превышающих 20 кратна трем")
else:
    print("Сумма чисел не превышающих 20 не кратна трем")
