a = int(input("Введите a: "))
b = int(input("Введите b: "))

if a != 0 and b != 0:
    if abs(a) % abs(b) == 0 or abs(b) % abs(a) == 0:
        print("Одно из чисел является делителем другого")
    else:
        print("Нет делителя")
else:
    print("Один из аргументов равен нулю, деление на ноль невозможно")
